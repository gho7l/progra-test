public void selectionSort() {
        int size = al.size();
        for(int i = 0; i <= size - 2; i++) {
            int posMin = i;
            for(int j = i + 1; j <= size - 1; j++)
                if(al.get(j).compareTo(al.get(posMin)) < 0) 
                    posMin = j;
            if(posMin != i)
                al.set(i, al.set(posMin, al.get(i)));
        }
    }
