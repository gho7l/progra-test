public void selectionSort() {
    int size = al.size();
    for(int i = 0; i <= size - 2; i++) {
        int posMin = i;
        for(int j = i + 1; j <= size - 1; j++)
            if(al.get(j).compareTo(al.get(posMin)) < 0) 
                posMin = j;
        if(posMin != i)
            al.set(i, al.set(posMin, al.get(i)));
    }
}
                    
                
                    
public void bubbleSort() {
     for(int i = 0; i < al.size(); i++)
         for(int j = 0; j < al.size() - i - 1; j++) 
             if(al.get(j).compareTo(al.get(j+1)) > 0) {
                 Long temp = al.get(j);
                 al.set(j, al.get(j+1);
                 al.set(j+1, temp);
             }
}
                    
                
                    
public void insertionSort() {
		for(int i = 1; i < al.size(); i++) {
			long n = al.get(i);
			int j = i - 1;
			while((j >= 0) && (n < al.get(j))) {
				al.set(j + 1, al.get(j));
		    	j--;
			}
     	al.set(j+1, n);
     }
}
                    
                
                    
public void collectionSort() {
	Collections.sort(al);
}
                    
                
                    
public void quickSort(int left, int right) {
	if(left < right) {
		int p = partition(left, right);
		quickSort(left, p-1);
		quickSort(p+1, right);
	}
}
		
public int partition(int left, int right) {
	int i = left;
	int j = right - 1;
	long pivot = al.get(right);
	while(i < j) {
    	while(i < right && al.get(i) <= pivot)
			i++;
		while(j > left && al.get(j) >= pivot) 
			j--;
		if(i < j) {
			long help = al.get(i);
			al.set(i, al.get(j));
			al.set(j, help);
		}
	}
	if(al.get(i) > pivot) {
		al.set(right, al.get(i));
		al.set(i, pivot);
	}
	return i;
}
