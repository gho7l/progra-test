SEARCH

LINEAR SEARCH

public int linearSearch(int start, int end, int n) {
    if(end < start)
        return -1;
    if(tab[start] == n)
        return start;
    if(tab[end] == n)
        return end;
    return linearSearch(start+1, end-1, n);
}

BINARY SEARCH

public int binarySearch(int start, int end, int n) {
    if(end >= start) {
        int mid = (start+end)/2;
        if(tab[mid] == n)
            return mid;
        if(tab[mid] > n)
            return binarySearch(start, mid-1, n);
        else
            return binarySearch(mid+1, end, n);
    }
}
