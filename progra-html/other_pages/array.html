ARRAY

private int[] tab = new int[size];
public static final int MAXSIZE = 100;
private int size = 0;
                    
public int getLength() {
    return tab.length;
}	
                    
public int get(int i) {
    if(size > MAXSIZE) 
        return null;
    return tab[i];
}
                    
public void add2End(int i) {
    if(size < MAXSIZE) 
        tab[size] = i++;
}
                    
public void remove(int i) {
    if(i < size) {
        for(int j = i; i < size-1; i++) 
            tab[i] == tab[i+1];
        size--;
    }
}
